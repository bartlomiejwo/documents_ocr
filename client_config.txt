1. instalacja debian
2. konfiguracja statycznego IP:
	sudo nano /etc/network/interfaces
	zakomentowujemy dhcp
	dodajemy:
		iface ens192 inet static
			address [ip_address]
			netmask [mask]
			gateway [gateway]
			dns-nameservers [dns1] [dns2] [dns3]

3. sudo apt install open-vm-tools

4. Instalacja ssh "sudo apt install openssh-server"

5. instalacja ftp "sudo apt install vsftpd"

6. Instalacja python3.8
	sudo apt install build-essential zlib1g-dev libncurses5-dev libgdbm-dev libnss3-dev libssl-dev libsqlite3-dev libreadline-dev libffi-dev curl libbz2-dev
	
	curl -O https://www.python.org/ftp/python/3.8.10/Python-3.8.10.tar.xz
	tar -xf Python-3.8.10.tar.xz
	cd Python-3.8.10
	./configure --enable-optimizations --enable-loadable-sqlite-extensions
	make -j 3 # 1.5x number of cores
	Modify the -j to correspond to the number of cores in your processor. You can find the number by typing nproc.

	sudo make altinstall
	Do not use the standard make install as it will overwrite the default system python3 binary.
	
7. mkdir /documents-archive/data

8. Umieszczenie plików projektu na serwerze w folderze /documents-archive

9. Stworzenie venv w /documents-archive/documents_ocr_client
	cd /documents-archive/documents_ocr_client
	python3.8 -m venv ./venv
	source venv/bin/activate
	pip install -r /documents-archive/documents_ocr_client/documents_ocr_client/requirements.txt
	
10. Utworzenie użytkownika do obsługi documents-archive
	sudo useradd documents-archive
	sudo passwd documents-archive	
	
11. Utworzenie katalogów:
	mkdir /documents-archive/data/logs
	mkdir /documents-archive/data/queue
	
12. Konfiguracja widoczności udostępnionego w sieci folderu archiwum z Windowsa:

	login: lgn
	password: psswd

	sudo apt install cifs-utils
	sudo mkdir /media/documents-archive
	sudo nano /etc/win-credentials
		do tego pliku wpisujemy dane uwierzytelniające:
			username=user
			password=password
			domain=domain

	sudo chown root: /etc/win-credentials
	sudo chmod 600 /etc/win-credentials

	sudo nano /etc/fstab
		dodajemy linię: //server/DOCUMENTS /media/documents-archive cifs uid=admin,gid=documents-archive,credentials=/etc/win-credentials,vers=1.0,file_mode=0775,dir_mode=0775

	Tworzymy symlink: sudo ln -s /media/documents-archive /documents-archive/data/documents-archive

13. Zmiana uprawnień do /documents-archive/data
	sudo chown admin:documents-archive /documents-archive/data/logs
	sudo chmod 775 /documents-archive/data/logs
	
	sudo chown admin:documents-archive /documents-archive/data/queue
	sudo chmod 775 /documents-archive/data/queue
	
14. Konfiguracja aplikacji

SERVER_IP = 192.168.1.3

LOGGING_PATH = /documents-archive/data/logs
QUEUE_PATH = /documents-archive/data/queue

+ konfiguracja pozostałych ścieżek windowsowych (incoming, processed, error, duplicates + processed dla każdego typu dokumentu)

15. Ustawienie działania aplikacji jako usługa

sudo nano /etc/systemd/system/documents_archive.service
	
[Unit]
Description=documents_archive
After=network.target
StartLimitIntervalSec=0

[Service]
User=documents-archive
Group=documents-archive
WorkingDirectory=/documents-archive/documents_ocr_client/documents_ocr_client
ExecStart=/documents-archive/documents_ocr_client/venv/bin/python3 /documents-archive/documents_ocr_client/documents_ocr_client/main.py
Restart=always
RestartSec=3

[Install]
WantedBy=multi-user.target

sudo systemctl enable documents_archive.service
sudo systemctl start documents_archive.service
sudo systemctl status documents_archive.service
sudo journalctl -u documents_archive.service
	
16. Dodanie do nagiosa

sudo apt install nagios-nrpe-server -y
sudo systemctl enable nagios-nrpe-server
sudo systemctl start nagios-nrpe-server

sudo nano /etc/default/nagios-nrpe-server # odkomentowujemy NRPE OPTS
sudo nano /etc/nagios/nrpe.cfg # ustawiamy konfiguracje
sudo systemctl restart nagios-nrpe-server
